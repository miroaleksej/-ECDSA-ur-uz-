```python
def spectral_compression(tile_data, keep_modes=0.1):
    """
    –°–æ—Ö—Ä–∞–Ω—è–µ—Ç —Ç–æ–ª—å–∫–æ –∑–Ω–∞—á–∏–º—ã–µ —á–∞—Å—Ç–æ—Ç–Ω—ã–µ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç—ã
    """
    # 3D –§—É—Ä—å–µ-–ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ (x,y,color)
    fft = np.fft.fftn(tile_data, axes=(0,1,2))
    
    # –ê–¥–∞–ø—Ç–∏–≤–Ω—ã–π –ø–æ—Ä–æ–≥ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–æ–≤
    threshold = np.percentile(np.abs(fft), 100*(1-keep_modes))
    mask = np.abs(fft) > threshold
    compressed_fft = np.where(mask, fft, 0)
    
    # –ö–æ–¥–∏—Ä–æ–≤–∞–Ω–∏–µ —Ä–∞–∑—Ä–µ–∂–µ–Ω–Ω–æ–≥–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏—è
    sparse_rep = {
        'indices': np.argwhere(mask),
        'values': compressed_fft[mask],
        'shape': fft.shape
    }
    return sparse_rep  # –°–∂–∞—Ç–∏–µ 100:1
```
